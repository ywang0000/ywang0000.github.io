# Recode any missing entries as NaN.
# The CPA column (cme_angle), replace 'Halo' with 'NaN'. Create a new 'is_halo' column
# The width column, remove any non-numeric part. Creeat a new 'width_lower_bound' column
# Combine date and time columns for start, end and cme so they can be encoded as datetime objects.


for i, row in nsa_df.iterrows():
   
   # replace missing start_frequency
   if row['start_frequency'] == '????':
       nsa_df.at[i,'start_frequency'] = "NaN"
    
   # replace missing end_frequency
   if row['end_frequency'] == '????':
       nsa_df.at[i,'end_frequency'] = "NaN"
    
   # replace missing flare_region
   if row['flare_region'] == '-----':
       nsa_df.at[i,'flare_region'] = "NaN"
    
   # replace missing flare_classification
   if row['flare_classification'] == '----':
       nsa_df.at[i,'flare_classification'] = "NaN"
    
   # The CPA column (cme_angle), replace 'Halo' with 'NaN'. Create a new 'is_halo' column
   if row['cme_angle'] == 'Halo':
       nsa_df.at[i,'cme_angle'] = "NaN"
       nsa_df.at[i,'is_halo'] = True
   elif row['cme_angle'] == '----':
       nsa_df.at[i,'cme_angle'] = "NaN"
       nsa_df.at[i,'is_halo'] = False
   else: 
       nsa_df.at[i,'is_halo'] = False
        
   # The width column, remove any non-numeric part. Creeat a new 'width_lower_bound' column
   if row['cme_width'].startswith('>'):
       nsa_df.at[i,'cme_width'] = row['cme_width'][1:]
       nsa_df.at[i,'width_lower_bound'] = True
   elif row['cme_width'] == '----':
       nsa_df.at[i,'cme_width'] = "NaN"
       nsa_df.at[i,'width_lower_bound'] = False
   else: 
       nsa_df.at[i,'width_lower_bound'] = False

   # Combine date and time columns for start, end and cme so they can be encoded as datetime objects.
   # Combine date and time columns for start   
   nsa_df.at[i,'start_time'] = pd.to_datetime(row['start_date'] + ' ' + row['start_time'])
    
   # if end_time is 24:00, replace it with 00:00 and increate date by one day 
   # Combine date and time columns for end   
   if row['end_time'] == '24:00':
       tmp = row['start_date'][0 : 5] + row['end_date'] + ' 00:00'
       dt = pd.to_datetime(tmp) + datetime.timedelta(days=1)
   else: 
       tmp = row['start_date'][0 : 5] + row['end_date'] + ' ' + row['end_time']
       dt = pd.to_datetime(tmp)
   nsa_df.at[i,'end_time'] = dt

   # replace missing cme_date
   # Combine date and time columns for cme   
   if row['cme_date'] == '--/--':
       nsa_df.at[i,'cme_time'] = "NaN"
   else: 
       tmp = row['start_date'][0 : 5] + row['cme_date'] + ' ' + row['cme_time']
       dt = pd.to_datetime(tmp)
       nsa_df.at[i,'cme_time'] = dt
        
nsa_df = nsa_df.drop('start_date', axis=1)
nsa_df = nsa_df.drop('end_date', axis=1)

# strategy
# 1. for each entry in SpaceWeatherLive table, find all NASA that has same classification category
# 2. find all the entries that have not been mapped to a SpaceWeatherLive entry
# 3. find the entry that has the nearest numeric value
# 4. if there are multiple entries, find the one that has nearest start time

# add a new column to record the mapped index of SpaceWeatherLive entry
nsavalid['mapindex'] = -1

# add a new column to record whether the row has been mapped to a SpaceWeatherLive entry
nsavalid['mapped'] = False

for i, row in top50_df.iterrows():

    # 1. for each entry in SpaceWeatherLive table, find all NASA that has same classification category
    nsafind=nsavalid.loc[nsavalid['category'] == row['x_classification'][0]]
    # 2. find all the entries that have not been mapped to a SpaceWeatherLive entry
    nsafind=nsafind.loc[nsafind['mapped'] == False ]

    # get the numeric part from 'x_classification'
    value = float(re.findall(r'[\d]+[.]?[\d]*', row['x_classification'])[0])
    
    # get the closest value among qualified nasa entries
    closevalue = min(nsafind['numeric_value'], key=lambda x:abs(x-value))
    
    # locate row(s) that match the 'closevalue' found in last step
    nsafind=nsafind.loc[nsafind['numeric_value'] == closevalue]
    
    # if multiple rows, find the one that has nearest start time
    if len(nsafind) > 1:
        closedate=min(nsafind.start_datetime, key=lambda x:abs(x - row['start_time']))
        nsafind = nsafind[nsafind['start_datetime'] == closedate]

    # determine the index of best match row
    id = nsafind.iloc[[0]].index.item()
    
    # record the index of best matched SpaceWeatherLive entry
    nsavalid.at[id,'mapindex'] = i
    
    # mark the row as mapped to SpaceWeatherLive entry
    nsavalid.at[id,'mapped'] = True
    
# create a DataFrame that contains only entries that have been mapped to SpaceWeatherLive    
nsavalid_mapped = nsavalid[nsavalid['mapindex'] != -1]
nsavalid_mapped = nsavalid_mapped.drop('category', axis=1)
nsavalid_mapped = nsavalid_mapped.drop('numeric_value', axis=1)
nsavalid_mapped = nsavalid_mapped.drop('mapped', axis=1)

        yoffset = row.mapindex * 10 if row.mapindex < 20 else row.mapindex * 6
